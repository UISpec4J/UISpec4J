1   package org.uispec4j.assertion.testlibrairies;

3   import java.util.ArrayList;
4   import java.util.List;
5   
6   enum TestLibraries {
7     JUNIT("junit.framework.Assert", JUnitLibrary.class),
8     TESTNG("org.testng.Assert", TestNGLibrary.class);
9   
10    private String representativeClassPath;
11    private TestLibrary dependency;
12  
13    private static final String TEST_LIBRARY_PROPERTY = "uispec4j.test.library";
14  
15    private TestLibraries(String assertPath, Class<? extends TestLibrary> dependencyClass) {
16      this.representativeClassPath = assertPath;
17      try {
18        this.dependency = dependencyClass.newInstance();
19      }
20      catch (Exception e) {
21        throw new RuntimeException(e);
22      }
23    }
24  
25    public boolean isPresent() {
26      ClassLoader classLoader = AssertAdapter.class.getClassLoader();
27      try {
28        classLoader.loadClass(representativeClassPath);
29        return true;
30      }
31      catch (ClassNotFoundException e) {
32        return false;
33      }
34    }
35  
36    public TestLibrary getDependency() {
37      return dependency;
38    }
39  
40    public static TestLibrary loadDependency() {
41      String library = System.getProperty(TEST_LIBRARY_PROPERTY);
42      if (library != null && library.length() > 0) {
43        return getImposedLibrary(library);
44      }
45      return retrieveDependency();
46    }
47  
48    private static TestLibrary retrieveDependency() {
49      TestLibraries[] candidates = values();
50      List<TestLibraries> libraries = new ArrayList<TestLibraries>();
51      for (TestLibraries candidate : candidates) {
52        if (candidate.isPresent()) {
53          libraries.add(candidate);
54        }
55      }
56      if (libraries.size() == 0) {
57        return new UISpecLibrary();
58      }
59  
60      if (libraries.size() > 1) {
61        return new InvalidLibrary("UISpec4J found several testing frameworks in your classpath. " +
62                                "You must set the '" + TEST_LIBRARY_PROPERTY +
63                                "' property to one among " +
64                                getList() + " or trim your classpath to keep only one of them.");
65      }
66      return libraries.get(0).getDependency();
67    }
68  
69    private static String getList() {
70      StringBuffer buffer = new StringBuffer("[");
71      TestLibraries[] libraries = values();
72      for (int i = 0; i < libraries.length; i++) {
73        String str = libraries[i].name().toLowerCase();
74        buffer.append(str);
75        buffer.append((i == libraries.length - 1) ? "]" : ", ");
76      }
77      return buffer.toString();
78    }
79  
80    private static TestLibrary getImposedLibrary(String libraryName) {
81      TestLibraries testLibrary = null;
82      try {
83        testLibrary = valueOf(libraryName.toUpperCase());
84      }
85      catch (IllegalArgumentException e) {
86        return new InvalidLibrary("You required UISpec4J to use " + libraryName + " as your testing framework, but UISpec4J does not recognize it. " +
87                                "Please use one of the following, before running your tests: " +
88                                getList());
89      }
90      if (!testLibrary.isPresent()) {
91        return new InvalidLibrary("You required UISpec4J to use " + libraryName + " as your testing framework. " +
92                                "Please add it to the classpath and retry.");
93      }
94      return testLibrary.getDependency();
95    }
