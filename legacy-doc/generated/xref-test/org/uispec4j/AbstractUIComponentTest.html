12    public void testGetComponentTypeName() throws Exception {
13      XmlAssert.assertEquivalent("<awtLabel name='myName'/>",
14                                 new DummyAwtUIComponent("myName").getDescription());
15    }
16  
17    private static final class DummyAwtUIComponent extends AbstractUIComponent {
18      private final Component component = new Label();
19  
20      DummyAwtUIComponent(String name) {
21        component.setName(name);
22      }
23  
24      public Component getAwtComponent() {
25        return component;
26      }
27  
28      public String getDescriptionTypeName() {
29        return "awtLabel";
30      }
31    }

33    public void testComponentUsesBlackAsDefaultForegroundColor() throws Exception {
34      final JComponent jComponent = new JComponent() {
35      };
36      AbstractUIComponent uiComponent = new AbstractUIComponent() {
37        public Component getAwtComponent() {
38          return jComponent;
39        }
40  
41        public String getDescriptionTypeName() {
42          return "";
43        }
44      };
45  
46      assertTrue(uiComponent.foregroundEquals("black"));
47    }

49    public void testForeground() throws Exception {
50      DummyAwtUIComponent uiComponent = new DummyAwtUIComponent("name");
51      uiComponent.component.setForeground(ColorUtils.getColor("2255F7"));
52  
53      assertTrue(uiComponent.foregroundEquals("2255F7"));
54      assertFalse(uiComponent.foregroundEquals("2255F8"));
55  
56      assertTrue(uiComponent.foregroundNear("2255F7"));
57      assertTrue(uiComponent.foregroundNear("2255F8"));
58      assertFalse(uiComponent.foregroundNear("FF5500"));
59  
60      assertTrue(uiComponent.foregroundNear("blue"));
61      assertFalse(uiComponent.foregroundNear("red"));
62    }

79    public void testGetContainer() throws Exception {
80      JPanel rootPanel = new JPanel();
81      rootPanel.setName("rootPanel");
82      JPanel middlePanel = new JPanel();
83      rootPanel.add(middlePanel);
84      JLabel jLabel = new JLabel();
85      middlePanel.add(jLabel);
86      TextBox label = new TextBox(jLabel);
87  
88      assertSame(middlePanel, label.getContainer().getAwtComponent());
89  
90      assertSame(rootPanel, label.getContainer("rootPanel").getAwtComponent());
91      assertNull(label.getContainer("unknown"));
92  
93      assertNull(new Panel(rootPanel).getContainer());
94    }
