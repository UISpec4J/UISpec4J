12  public abstract class PanelComponentFinderTestCase extends UnitTestCase {
13    protected JPanel jPanel;
14    protected Panel panel;
15    protected List components = new ArrayList();
16  
17    protected void setUp() throws Exception {
18      super.setUp();
19      jPanel = new JPanel();

22    }
23  
24    protected <T extends Component> T addComponent(Class<T> awtComponentclass, String name) throws Exception {
25      T component = createComponent(awtComponentclass, name);
26      components.add(component);
27      jPanel.add(component);
28      return component;
29    }
30  
31    private <T extends Component> T createComponent(Class<T> awtComponentclass, String name) throws Exception {
32      Component component;
33      try {
34        Constructor constructor = awtComponentclass.getConstructor(new Class[]{String.class});
35        component = (Component) constructor.newInstance(new Object[]{name});
36      }
37      catch (NoSuchMethodException e) {
38        component = (Component) awtComponentclass.newInstance();
39      }
40      component.setName(name);
41      return (T)component;
42    }
