15    public void testWindowManagesMenuBars() throws Exception {
16      Window window = new Window(new Frame());
17      try {
18        window.getMenuBar();
19        throw new AssertionFailureNotDetectedError();
20      }
21      catch (AssertionError e) {
22        assertEquals("This component has no menu bar", e.getMessage());
23      }
24    }
25  
26    public void testGetTitle() throws Exception {
27      assertEquals("", createWindow().getTitle());
28    }
29  
30    public void testAssertTitleEquals() throws Exception {
31      final Window window = createWindow();
32      assertTrue(window.titleEquals(""));
33      checkAssertionFails(window.titleEquals("title"),
34                          "Unexpected title - expected:<[title]> but was:<[]>");
35    }
36  
37    public void testAssertTitleContains() throws Exception {
38      final Window window = createWindow();
39      assertTrue(window.titleContains(""));
40      checkAssertionFails(window.titleContains("title"),
41                          "expected to contain:<title> but was:<>");
42    }
43  
44    public void testGetDescription() throws Exception {
45      Window window = createWindow();
46      window.getAwtComponent().setName("myFrame");
47  
48      JTextField textField = new JTextField();
49      textField.setName("myText");
50      addComponent(window, textField);
51  
52      XmlAssert.assertEquivalent("<window title=''>" +
53                                 "  <textBox name='myText'/>" +
54                                 "</window>",
55                                 window.getDescription());
56    }
