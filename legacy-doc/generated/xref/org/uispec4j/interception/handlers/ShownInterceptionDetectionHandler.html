3   import org.uispec4j.Window;
4   import org.uispec4j.assertion.testlibrairies.AssertAdapter;
5   
6   public class ShownInterceptionDetectionHandler extends AbstractInterceptionHandlerDecorator {
7     public static final String NO_WINDOW_WAS_SHOWN_ERROR_MESSAGE = "No window was shown (timeout expired)";
8   
9     private boolean windowWasShown = false;
10    private long waitTimeLimit;
11  
12    public ShownInterceptionDetectionHandler(InterceptionHandler handler, long waitTimeLimit) {
13      super(handler);
14      this.waitTimeLimit = waitTimeLimit;
15    }
16  
17    public void process(Window window) {
18      synchronized (this) {
19        windowWasShown = true;
20        notify();
21        super.process(window);
22      }
23    }
24  
25    public void waitWindow() {
26      long end = System.currentTimeMillis() + waitTimeLimit;
27      long still = waitTimeLimit;
28      synchronized (this) {
29        while (!windowWasShown && still > 0) {
30          try {
31            wait(still);
32            still = end - System.currentTimeMillis();
33          }
34          catch (InterruptedException e) {
35          }
36        }
37      }
38      if (!windowWasShown) {
39        AssertAdapter.fail(NO_WINDOW_WAS_SHOWN_ERROR_MESSAGE);
40      }
41    }
42  }
