9   public class CollectionComponentMatchersTest extends PanelComponentFinderTestCase {
10    private JButton component1;
11    private JButton component2;
12    private JButton component3;
13    private JTextField component4;
14  
15    protected void setUp() throws Exception {
16      super.setUp();
17      component1 = addComponent(JButton.class, "some text");
18      component2 = addComponent(JButton.class, "other text");
19      component3 = addComponent(JButton.class, "nothing");
20      component4 = addComponent(JTextField.class, "last text");
21    }
22  
23    public void testIntersectionMatcher() throws Exception {
24      ComponentMatcher matcher = and(displayedNameSubstring("text"), fromClass(JButton.class));
25  
26      TestUtils.assertSwingComponentsEquals(new JButton[]{component1, component2},
27                                            panel.getSwingComponents(matcher));
28    }
29  
30    public void testUnionComponentMatcher() throws Exception {
31      ComponentMatcher matcher = or(displayedNameSubstring("text"), fromClass(JButton.class));
32  
33      TestUtils.assertSwingComponentsEquals(new Component[]{component1, component2, component3, component4},
34                                            panel.getSwingComponents(matcher));
35    }
36  
37    public void testNegatedComponentMatcher() throws Exception {
38      ComponentMatcher matcher = ComponentMatchers.not(displayedNameSubstring("text"));
39      TestUtils.assertSwingComponentsEquals(new Component[]{component3},
40                                            panel.getSwingComponents(matcher));
41    }
42  }
