1   package org.uispec4j;

3   import org.uispec4j.assertion.Assertion;
4   import org.uispec4j.utils.Utils;
5   
6   import javax.swing.*;
7   import java.util.List;
8   
9   /**
10   * Wrapper for JSpinner components implementing a SpinnerListModel.
11   */
12  public class ListSpinner extends Spinner {
13    private SpinnerListModel model;
14  
15    public ListSpinner(JSpinner jSpinner) {
16      super(jSpinner);
17      SpinnerModel model = jSpinner.getModel();
18      if (!SpinnerListModel.class.isInstance(model)) {
19        throw new ItemNotFoundException("Expected JSpinner using a SpinnerListModel");
20      }
21      this.model = (SpinnerListModel)model;
22    }
23  
24    /**
25     * Checks the list spinner contents
26     */
27    public Assertion contentEquals(final Object... expectedContents) {
28      return new Assertion() {
29        public void check() {
30          List list = model.getList();
31          Utils.assertEquals(expectedContents, list.toArray(new Object[list.size()]));
32        }
33      };
34    }
35  }
