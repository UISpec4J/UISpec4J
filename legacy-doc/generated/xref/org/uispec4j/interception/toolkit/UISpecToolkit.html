10  /**
11   * Mock toolkit used for intercepting displayed frames and dialogs.<p>
12   * You can set it up by calling the {@link #setUp()} method.
13   *
14   * @see <a href="http://www.uispec4j.org/intercepting-windows">Intercepting windows</a>
15   */
16  public class UISpecToolkit extends ToolkitDelegate {
17    static final String SYSTEM_PROPERTY = "awt.toolkit";
18    static final String UNIX_SYSTEM_DEFAULT_VALUE = "sun.awt.motif.MToolkit";
19    static final String WINDOWS_SYSTEM_DEFAULT_VALUE = "sun.awt.windows.WToolkit";
20  
21    private static String awtToolkit;
22  
23    public UISpecToolkit() {
24      setUp();
25    }
26  
27    /**
28     * @see UISpec4J#init
29     * @deprecated Do not call this one directly anymore - use {@link UISpec4J#init} instead
30     */
31    public static void setUp() {
32      if (underlyingToolkit != null) {
33        return;
34      }
35      awtToolkit = System.getProperty(SYSTEM_PROPERTY);
36      if (awtToolkit == null) {
37        setAwtToolkitProperty();
38      }
39      buildUnderlyingToolkit(awtToolkit);
40      System.setProperty(SYSTEM_PROPERTY, UISpecToolkit.class.getName());
41    }
42  
43    /**
44     * Sets the <code>awt.toolkit</code> to its initial value.
45     * <p>This method will only work properly if the toolkit has not yet been instanciated by Swing.
46     */
47    public static void restoreAwtToolkit() {
48      System.setProperty(SYSTEM_PROPERTY, awtToolkit);
49    }
50  
51    public static UISpecToolkit instance() {
52      Toolkit defaultToolkit = Toolkit.getDefaultToolkit();
53      if (!(defaultToolkit instanceof UISpecToolkit)) {
54        fail("You must call UISpec4J.init() before using it");
55      }
56      return (UISpecToolkit)defaultToolkit;
57    }
58  
59    protected LightweightPeer createComponent(Component target) {
60      if (target instanceof JPopupMenu) {
61        UISpecDisplay.instance().setCurrentPopup((JPopupMenu)target);
62      }
63      return super.createComponent(target);
64    }
65  
66    public FramePeer createFrame(Frame target) {
67      return new UISpecFramePeer(target);
68    }
69  
70    public DialogPeer createDialog(Dialog target) throws HeadlessException {
71      if (!(target instanceof JDialog)) {
72        throw new InterceptionInternalError("Dialogs of type '"
73                                            + target.getClass().getName() + "' are not supported.");
74      }
75      return new UISpecDialogPeer((JDialog)target);
76    }
77  
78    public WindowPeer createWindow(Window target) throws HeadlessException {
79      return new UISpecWindowPeer(target);
80    }
81  
82    public CanvasPeer createCanvas(Canvas target) {
83      return Empty.NULL_CANVAS_PEER;
84    }
85  
86    public PanelPeer createPanel(Panel target) {
87      return Empty.NULL_PANEL_PEER;
88    }
89  
90    public RobotPeer createRobot(Robot robot, GraphicsDevice device) throws AWTException, HeadlessException {
91      return Empty.NULL_ROBOT;
92    }
93  
94    public KeyboardFocusManagerPeer createKeyboardFocusManagerPeer(KeyboardFocusManager keyboardFocusManager) throws HeadlessException {
95      return null;
96    }
97  
98    protected boolean syncNativeQueue(long l) {
99      return false;
100   }
101

103     return 0;
104   }
105

107     return 0;
108   }
109 
110   protected MouseInfoPeer getMouseInfoPeer() {
111     return Empty.NULL_MOUSE_INFO;
112   }
113 
114   private static void fail(String msg) {
115     throw new InterceptionInternalError(msg);
116   }
117 
118   private static void setAwtToolkitProperty() {
119     try {
120       Class.forName(WINDOWS_SYSTEM_DEFAULT_VALUE);
121       awtToolkit = WINDOWS_SYSTEM_DEFAULT_VALUE;
122     }
123     catch (ClassNotFoundException e) {
124       try {
125         Class.forName(UNIX_SYSTEM_DEFAULT_VALUE);
126         awtToolkit = UNIX_SYSTEM_DEFAULT_VALUE;
127       }
128       catch (ClassNotFoundException e1) {
129         throw new AWTError("Unable to locate AWT Toolkit");
130       }
131     }
132   }
133 
134   private static void buildUnderlyingToolkit(String awtToolkit) {
135     try {
136       underlyingToolkit = (Toolkit)Class.forName(awtToolkit).newInstance();
137     }
138     catch (Exception e) {
139       throw new AWTError("Unable to load AWT Toolkit: " + awtToolkit + " - "
140                          + e.getLocalizedMessage());
141     }
142   }
143 
144   public InputMethodDescriptor getInputMethodAdapterDescriptor() throws AWTException {
145     return null;
146   }
