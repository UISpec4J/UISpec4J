8   public class AbstractSwingUIComponentTest extends TestCase {

10    public void testTooltips() throws Exception {
11      DummySwingUIComponent component = new DummySwingUIComponent("label");
12      component.getAwtComponent().setToolTipText("Hello world!");
13  
14      UISpecAssert.assertThat(component.tooltipEquals("Hello world!"));
15      UISpecAssert.assertFalse(component.tooltipEquals("hello world!"));
16      UISpecAssert.assertFalse(component.tooltipEquals("Hello world"));
17  
18      UISpecAssert.assertThat(component.tooltipContains("world"));
19      UISpecAssert.assertFalse(component.tooltipContains("something else"));
20    }
21  
22    private static final class DummySwingUIComponent extends AbstractSwingUIComponent {
23      private final JComponent component = new JLabel();
24  
25      DummySwingUIComponent(String name) {
26        component.setName(name);
27      }
28  
29      public JComponent getAwtComponent() {
30        return component;
31      }
32
