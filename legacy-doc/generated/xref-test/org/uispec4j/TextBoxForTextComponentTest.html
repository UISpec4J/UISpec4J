396   private void checkClickOnHyperlink(String html, String link, String expectedTarget) throws Exception {
397     JTextPane textPane = createTextPane(html);
398     DummyHyperlinkListener listener = new DummyHyperlinkListener();
399     textPane.addHyperlinkListener(listener);
400     TextBox textComponent = new TextBox(textPane);
401     textComponent.clickOnHyperlink(link);
402     assertEquals(1, listener.getCallCount());
403     assertEquals(expectedTarget, listener.getLastEvent().getDescription());
404 
405     listener.reset();
406     textComponent.triggerClickOnHyperlink(link).run();
407     assertEquals(1, listener.getCallCount());
408     assertEquals(expectedTarget, listener.getLastEvent().getDescription());
409   }

426   private static class DummyHyperlinkListener implements HyperlinkListener {
427     int callCount;
428     HyperlinkEvent lastEvent;
429 
430     public void hyperlinkUpdate(HyperlinkEvent event) {
431       callCount++;
432       this.lastEvent = event;
433     }
434 
435     public int getCallCount() {
436       return callCount;
437     }
438 
439     public HyperlinkEvent getLastEvent() {
440       return lastEvent;
441     }
442 
443     public void reset() {
444       callCount = 0;
445       lastEvent = null;
446     }
447   }
